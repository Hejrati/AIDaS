conda create -n oct-segmenter-env python=3.10 

conda activate oct-segmenter-env


pip install oct_segmenter-0.8.4-py3-none-any.whl


pip install oct_image_segmentation_models-0.8.4-py3-none-any.whl



prediction:

oct-segmenter predict -c %1 -m %2 -i %3 

%1 -- path and filename of config.json 
%2 -- path and name of model file
%3 -- path and name of .tiff file for segmentation.  Default will place output results in the same folder.  Use -o option to specify another folder.

use -d option to segment all .tiff files in the folder.
